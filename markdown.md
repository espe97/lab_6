**–û–ø–∏—Å–∞–Ω–∏–µ –∞–ª–≥–æ—Ä–∏—Ç–º–∞:**

from math import gcd

def pollards_rho(n, f=lambda x: (x**2 + 1) % n):
    """
    –†–µ–∞–ª–∏–∑–∞—Ü–∏—è –∞–ª–≥–æ—Ä–∏—Ç–º–∞ –ü–æ–ª–ª–∞—Ä–¥–∞ –¥–ª—è —Ñ–∞–∫—Ç–æ—Ä–∏–∑–∞—Ü–∏–∏ —á–∏—Å–ª–∞.
    
    –ü–∞—Ä–∞–º–µ—Ç—Ä—ã:
        n (int): –°–æ—Å—Ç–∞–≤–Ω–æ–µ —á–∏—Å–ª–æ –¥–ª—è —Ä–∞–∑–ª–æ–∂–µ–Ω–∏—è –Ω–∞ –º–Ω–æ–∂–∏—Ç–µ–ª–∏.
        f (function): –§—É–Ω–∫—Ü–∏—è –¥–ª—è –∏—Ç–µ—Ä–∞—Ü–∏–∏, –ø–æ —É–º–æ–ª—á–∞–Ω–∏—é f(x) = (x^2 + 1) mod n.
        
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç:
        int: –ù–µ—Ç—Ä–∏–≤–∏–∞–ª—å–Ω—ã–π –¥–µ–ª–∏—Ç–µ–ª—å —á–∏—Å–ª–∞ n –∏–ª–∏ None, –µ—Å–ª–∏ —Ñ–∞–∫—Ç–æ—Ä–∏–∑–∞—Ü–∏—è –Ω–µ —É–¥–∞–ª–∞—Å—å.
    """
    c = 1  # –ù–∞—á–∞–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ
    a = c
    b = c
    while True:
        a = f(a)  # –û–¥–Ω–æ –æ–±–Ω–æ–≤–ª–µ–Ω–∏–µ a
        b = f(f(b))  # –î–≤–∞ –æ–±–Ω–æ–≤–ª–µ–Ω–∏—è b
        d = gcd(abs(a - b), n)  # –ù–û–î(|a - b|, n)
        if d == n:  # –ù–µ—É–¥–∞—á–∞, –Ω—É–∂–Ω–æ –ø–µ—Ä–µ–∑–∞–ø—É—Å—Ç–∏—Ç—å —Å –Ω–æ–≤—ã–º c
            return None
        if d > 1:  # –£—Å–ø–µ—Ö, –Ω–∞–π–¥–µ–Ω –Ω–µ—Ç—Ä–∏–≤–∏–∞–ª—å–Ω—ã–π –¥–µ–ª–∏—Ç–µ–ª—å
            return d

# –ü—Ä–∏–º–µ—Ä: –§–∞–∫—Ç–æ—Ä–∏–∑–∞—Ü–∏—è —á–∏—Å–ª–∞ 1359331
n = 1359331

if factor:
    print(f"–ù–µ—Ç—Ä–∏–≤–∏–∞–ª—å–Ω—ã–π –¥–µ–ª–∏—Ç–µ–ª—å —á–∏—Å–ª–∞ {n}: {factor}.")
    print(f"–î—Ä—É–≥–æ–π –¥–µ–ª–∏—Ç–µ–ª—å: {n // factor}.")
else:
    print(f"–ù–µ —É–¥–∞–ª–æ—Å—å –Ω–∞–π—Ç–∏ –¥–µ–ª–∏—Ç–µ–ª—å –¥–ª—è —á–∏—Å–ª–∞ {n}.")
